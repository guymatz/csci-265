\begin{enumerate}

\item \textbf{Describe the languages the following CFGs generate:}

\begin{enumerate}
  \item
    \begin{equation*}
      \begin{aligned}
        S &\rightarrow aX \\
        X &\rightarrow aX | bX | \epsilon
      \end{aligned}
    \end{equation*}
    Words that start with $a$
  \item
    \begin{equation*}
      \begin{aligned}
        S &\rightarrow XaXaX \\
        X &\rightarrow aX | bX | \epsilon
      \end{aligned}
    \end{equation*}
    A string of $a$'s and $b$'s repeated 3 times, with 
    an $a$ in the middle of each repeat
  \item
    \begin{equation*}
      \begin{aligned}
        S &\rightarrow SS | XXX \\
        X &\rightarrow aX | Xa | b
      \end{aligned}
    \end{equation*}
    3x $a$'s and 3x $b$'s
\end{enumerate}

\item \textbf{Write CFGs for each of the languages defined by the following regular expressions:}
\begin{enumerate}
  \item $ab\kleene$
    \begin{equation*}
      \begin{aligned}
        S &\rightarrow aX \\
        X &\rightarrow bX | \epsilon
      \end{aligned}
    \end{equation*}

  \item $a\kleene b\kleene$
    \begin{equation*}
      \begin{aligned}
        S &\rightarrow XY \\
        X &\rightarrow aX | \epsilon \\
        Y &\rightarrow bY | \epsilon
      \end{aligned}
    \end{equation*}
    
    \item $(baa + abb) \kleene$
    \begin{equation*}
      \begin{aligned}
        S &\rightarrow baaS | abbS | \epsilon \\
      \end{aligned}
    \end{equation*}
  \item $ab (bb \kleene + aab ) \kleene$
    \begin{equation*}
      \begin{aligned}
        S &\rightarrow abX | \epsilon \\
        X &\rightarrow bbX | aabX | \epsilon \\
      \end{aligned}
    \end{equation*}
\end{enumerate}
